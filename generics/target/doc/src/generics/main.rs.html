2// Generic - Type of the argument/parameter
3// for example
4// below function just simply returns the i32 value that is passed
5/// Identity with return value as i32
6fn identity_i32(value: i32) -> i32 {
7    value
8}
9// What if I need the same function which accepts f64 parameter - I just need to create another function
10/// Identity with return value as i64
11fn identity_f64(value: f64) -> f64 {
12    value
13}

15// This makes tough time for us to create multiple functions for multiple types
16// But generic solves the problem
17
18/// `identity<T>` which has generic type
19///  Breakdown of the above code
20/// Generic is a place holder for the Type
21/// We specify <T> as a generic type near function name 
22/// We specify the same T to the type of the value 
23/// Finally we returns the same T as return type
24/// This will accept any type of value to be used as value parameter while invoking them
25/// Note that T is common community convention- we can use any Name there for the generic
26fn identity<T>(value: T) -> T {
27    value
28}

31/// function may have multiple arguments
32fn make_tuple(first: i32, second: f64) -> (i32, f64) {
33    (first, second)
34}// This function creates tuple with two elements of i32 first and f64 type as second element
35
36// lets recreate it with Generic of first element
37/// Multiple argument Generics - function may have multiple arg with one generic
38fn make_tuple_with_generic<T>(first: T, second: f64) -> (T, f64) {
39    (first, second)
40} // This function accepts any type of data in first parameter as it is T generic type. 
41// So we specify T whereever we use first parameter also in return 
42
43// Make both the parameters generic type
44/// Multiple Generics - function may have multiple generic
45fn make_tuple_both_generic<T, U>(first: T, second: U) -> (T, U) {
46    (first, second)
47}// We cant specify T for both the genric as rust will not able to allow as T can be string at first and i32 at second parameter
48// So we must use the second common community convention letter U. 
49// This covers both elements in any type

126/// Struct with Generic
127#[derive(Debug)]
128struct TreasureChest<T> {
129    captain: String,
130    treasure: T,
131}
132
133// We specify this method with generic for type String alone
134// This means this impl is specifically for String datatype
135impl TreasureChest<String> {
136    fn clean_treasure(&mut self){
137        self.treasure = self.treasure.trim().to_string();
138    }
139}
140
141/// Specify the generic type in impl
142impl TreasureChest<[&str; 4]> {
143    fn count_number_of_treasure(&self) -> usize{
144        self.treasure.len()
145    }
146}
